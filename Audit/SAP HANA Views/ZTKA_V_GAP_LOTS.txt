CREATE VIEW [ZTKA_V_GAP_LOTS] AS 
SELECT H."DocType" AS "TransType",
		H."ApplyEntry" AS "DocEntry",
		H."ApplyLine" AS "DocLineNum",
		H."LocCode",
		H."ItemCode",
		B."DistNumber" AS "Lot",
		SUM(L."Quantity") AS "Quantity"
FROM OITL H
INNER JOIN ITL1 L ON H."LogEntry" = L."LogEntry"
INNER JOIN OBTN B ON B."SysNumber" = L."SysNumber" AND B."AbsEntry" = L."MdAbsEntry"
GROUP BY H."DocType",
		H."ApplyEntry",
		H."ApplyLine",
		H."LocCode",
		H."ItemCode",
		B."DistNumber"
